流程
----
1.系統產生一四位數  
2.輸入一四位數  
3.比較  
{if(4A)->結束；否則->顯示_A_B，回到(2.)}  

TIP:  
四位數(A)--數字不重複  
個(a1)=(A%100)/10  
十(a2)=(A%1000)/100  
百(a3)=(A/100)%10  
千(a4)=A%1000  


程式碼
----
```java
public class Main {

    public static void main(String[] args) {
	// write your code here
        int[] iAr={0,0,0,0};//輸入值儲存{千,百,十,個}
        int[] pwdAr={0,0,0,0};;//亂數儲存{千,百,十,個}
        int  i,pwd,cntGuess=0;//宣告變數
        boolean b;//判斷位數內數字是否相同

        do {
            pwd = (int) (Math.random() * (10000));//取得亂數pwd

            for(int x=0;x<=3;x++){
                pwdAr[x]=NumberTurnArray(x,pwd);//亂數切割位數
            }
            
            b=CheckNumberSame( pwdAr[0], pwdAr[1], pwdAr[2], pwdAr[3]);//判斷數字是否相同
        }while((pwd/1000)==0||b);//當pwd<1000，重新產生一亂數

        java.util.Scanner sc=new java.util.Scanner(System.in);//建立Scanner物件
        do{

            System.out.print("輸入四位數(數字不重複)=");//印出輸入提示
            i=sc.nextInt();//取得輸入值

            cntGuess++;//猜測次數

            for(int x=0;x<=3;x++){
                iAr[x]=NumberTurnArray(x,i);//輸入值切割位數
            }
            b=CheckNumberSame( iAr[0], iAr[1], iAr[2], iAr[3]);//判斷數字是否相同
            if(b){//數字相同
                System.out.println("輸入值位數有相同數字!!");//印出數字相同訊息
            }
            CheckAB(pwdAr[0], pwdAr[1], pwdAr[2], pwdAr[3],iAr[0], iAr[1], iAr[2], iAr[3]);//確認輸入值幾A幾B


        }while(i!=pwd||b);//當輸入值=亂數且數字無相同，離開迴圈

        System.out.println("猜對了，你猜了"+cntGuess+"次");//印出正確和猜測次數訊息

    }

    static int NumberTurnArray(int a,int i) {
        if (a == 0) {//千位數
            return i / 1000;
        } else if (a == 1) {//百位數
            return (i / 100) % 10;
        } else if (a == 2) {//十位數
            return (i % 100) / 10;
        } else if (a == 3) {//個位數
            return i%10;
        }
        return 0;
    }

    static boolean CheckNumberSame(int a,int b,int c,int d){//取得傳遞值
        if(a==b || a==c || a==d || b==c || b==d || c==d){//判斷數字是否相同
            return true;//數字有相同
        }else{
            return false;//數字無相同
        }
    }

    static void CheckAB(int pwd0,int pwd1,int pwd2,int pwd3,int i0,int i1,int i2,int i3){//取得傳遞值
        int cntA=0,cntB=0;//宣告變數
            if(pwd0==i0){//千位數 數字正確 位置正確
                cntA++;
            }
            if(pwd1==i1){//百位數 數字正確 位置正確
                cntA++;
            }
            if(pwd2==i2){//十位數 數字正確 位置正確
                cntA++;
            }
            if(pwd3==i3){//個位數 數字正確 位置正確
                cntA++;
            }
            if(pwd0==i1 || pwd0==i2 || pwd0==i3){//千位數 數字正確 位置錯誤
                cntB++;
            }
            if(pwd1==i0 || pwd1==i2 || pwd1==i3){//百位數 數字正確 位置錯誤
                cntB++;
            }
            if(pwd2==i0 || pwd2==i1 || pwd2==i3){//十位數 數字正確 位置錯誤
                cntB++;
            }
            if(pwd3==i0 || pwd3==i1 || pwd3==i2){//個位數 數字正確 位置錯誤
                cntB++;
            }
        System.out.println(cntA+"A"+cntB+"B");//印出幾A幾B
    }
}
```
