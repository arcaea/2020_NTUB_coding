class Main {
  public static void main(String[] args) {
    float x,f_of_x;//建立變數

    java.util.Scanner sc=new java.util.Scanner(System.in);//建立Scanner物件
    System.out.print("x=");//f(x)的二次項
    x=sc.nextFloat();//取得x

    if(x<=-5){
      f_of_x=2*x;
    }else if(x>(-5) && x<=10){
      f_of_x=x-10;
    }else if(x>10 && x<=100){
      f_of_x=10+x;
    }else{//x>100
      f_of_x=100-2*x;
    }

    System.out.println(f_of_x);//輸出f(x)
  }
}
