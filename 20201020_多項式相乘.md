
f(x)=a2 X^2 + a1 X +a0
g(x)=b2 X^2 + b1 X +b0
輸入a2,a1,a0,b2,b1,b0,求c4,c3,c2,c1,c0?
----
```java
public class Main {

    public static void main(String[] args) {
        java.util.Scanner sc=new java.util.Scanner(System.in);
	    int[] a=new int[3];
	    int[] b=new int[3];
        int[] c=new int[5];

        System.out.println("inputA2");//a2是f(x)二次項
        a[2] = sc.nextInt();
        System.out.println("inputA1");//a1是f(x)一次項
        a[1] = sc.nextInt();
        System.out.println("inputA0");//a0是f(x)常數項
        a[0] = sc.nextInt();
        System.out.println("inputb2");//b2是g(x)二次項
        b[2] = sc.nextInt();
        System.out.println("inputb1");//b1是g(x)一次項
        b[1] = sc.nextInt();
        System.out.println("inputb0");//b0是g(x)常數項
        b[0] = sc.nextInt();

        /*
        c[4]=a[2]*b[2];
        c[3]=(a[2]*b[1]+a[1]*b[2]);
        c[2]=(a[2]*b[0]+a[1]*b[1]+a[2]*b[2]);
        c[1]=(a[1]*b[0]+a[2]*b[1]);
        c[0]=a[0]*b[0];
        */
        for(int i=0; i<3;i++){
            for(int j=0; j<3 ;j++) {
                c[i + j] = c[i + j] + a[i] * b[i];

            }
        }

        c[4]=a[2]*b[2];
        c[3]=(a[2]*b[1]+a[1]*b[2]);
        c[2]=(a[2]*b[0]+a[1]*b[1]+a[2]*b[2]);
        c[1]=(a[1]*b[0]+a[2]*b[1]);
        c[0]=a[0]*b[0];
        System.out.println("c4="+c[4]+"c3="+c[3]+"c2="+c[2]+"c1="+c[1]+"c0="+c[0]);

    }
}
```
