class Main {
  public static void main(String[] args) {
    int[] a=new int[100];//儲存資料
    int n=0;//陣列欄位
    for(;;){//建立無線迴圈
      if(n>=100) break;//限制迴圈
      if(n==0 || n==1){//建立初始值
        a[n]=1;
        n++;
        continue;//繼續迴圈
      }
      a[n]=a[n-1]+a[n-2];//計算欄位值
      n++;
    }
    for(int i=0;i<a.length;i++){
      System.out.println("a["+(i+1)+"]="+a[i]);
    }
  }
}
