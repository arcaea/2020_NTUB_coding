法1(無限迴圈)
```java
class Main {
  public static void main(String[] args) {
    
    int r = 0,pwd,max=99,min=1;//宣告變數
    r = (int)(Math.random()*97)+2;//2-98之間的亂數
    
    java.util.Scanner sc=new java.util.Scanner(System.in);//建立Scanner物件
    
    do{
      System.out.print("輸入"+min+"-"+max+"之間的一個整數=");
      pwd=sc.nextInt();//取得輸入值

      if(pwd<r && min<pwd){//當輸入值小於亂數值，且大於最小值
        min=pwd;
      }else if(pwd>r && max>pwd){//當輸入值大於亂數值，且小於最大值
        max=pwd;
      }else if(pwd==r){//當輸入值等於亂數值
        System.out.print("你猜到了!!!!");
        break;//終止迴圈
      }else{//不是1-99的數值
        continue;
      }
    }while(true);//無限迴圈
  }
}
```
